```sql
-- Day 1, Day 7, Day 30 retention
WITH cohorts AS (
  SELECT 
    user_id,
    DATE(created_at) as cohort_date,
    created_at
  FROM users
)
SELECT 
  cohort_date,
  COUNT(DISTINCT user_id) as cohort_size,
  COUNT(DISTINCT CASE 
    WHEN EXISTS (
      SELECT 1 FROM transactions t 
      WHERE t.user_id = cohorts.user_id 
      AND DATE(t.created_at) = cohort_date + 1
    ) THEN user_id 
  END) * 100.0 / COUNT(DISTINCT user_id) as day1_retention,
  COUNT(DISTINCT CASE 
    WHEN EXISTS (
      SELECT 1 FROM transactions t 
      WHERE t.user_id = cohorts.user_id 
      AND DATE(t.created_at) = cohort_date + 7
    ) THEN user_id 
  END) * 100.0 / COUNT(DISTINCT user_id) as day7_retention
FROM cohorts
GROUP BY cohort_date
ORDER BY cohort_date DESC
LIMIT 30;
```
