def process_withdrawal_economical(telegram_id, amount, wallet_address):
    """–≠–∫–æ–Ω–æ–º–Ω—ã–π withdrawal - —Å–æ–∑–¥–∞—ë–º —Ç–æ–ª—å–∫–æ net amount"""
    
    # 1. –ü—Ä–æ–≤–µ—Ä–∫–∏
    user = get_user_data(telegram_id)
    balance = user['tama']
    
    if balance < amount:
        return {'error': 'Insufficient balance'}
    
    # 2. –†–∞—Å—Å—á–∏—Ç–∞—Ç—å fee
    fee = int(amount * 0.05)
    net_amount = amount - fee
    
    # –†–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ fee (–í–ò–†–¢–£–ê–õ–¨–ù–û!)
    burn_amount = int(fee * 0.60)   # 300
    pool_amount = int(fee * 0.30)   # 150
    team_amount = fee - burn_amount - pool_amount  # 50
    
    # 3. Mint –¢–û–õ–¨–ö–û net amount (–≠–ö–û–ù–û–ú–ò–Ø!)
    result = subprocess.run([
        'spl-token', 'transfer',
        TAMA_MINT,
        str(net_amount),  # ‚Üê –¢–û–õ–¨–ö–û 9,500!
        wallet_address,
        '--fee-payer', PAYER_KEYPAIR,
        '--owner', PAYER_KEYPAIR,
        '--output', 'json'
    ], capture_output=True, text=True)
    
    if result.returncode != 0:
        return {'error': result.stderr}
    
    tx_data = json.loads(result.stdout)
    signature = tx_data.get('signature')
    
    # 4. –û–±–Ω–æ–≤–∏—Ç—å –±–∞–∑—É (—Å–ø–∏—Å–∞—Ç—å –í–°–Æ —Å—É–º–º—É)
    supabase.table('leaderboard').update({
        'tama': balance - amount,  # ‚Üê –°–ø–∏—Å–∞—Ç—å 10,000
        'total_withdrawn': user.get('total_withdrawn', 0) + net_amount
    }).eq('telegram_id', telegram_id).execute()
    
    # 5. –ó–∞–ø–∏—Å–∞—Ç—å —Ç—Ä–∞–Ω–∑–∞–∫—Ü–∏—é
    supabase.table('tama_economy').insert({
        'telegram_id': telegram_id,
        'transaction_type': 'withdrawal',
        'amount': -amount,
        'net_amount': net_amount,
        'fee': fee,
        'burn_amount': burn_amount,  # ‚Üê –í–∏—Ä—Ç—É–∞–ª—å–Ω–æ!
        'pool_amount': pool_amount,  # ‚Üê –í–∏—Ä—Ç—É–∞–ª—å–Ω–æ!
        'team_amount': team_amount,  # ‚Üê –í–∏—Ä—Ç—É–∞–ª—å–Ω–æ!
        'signature': signature,
        'wallet_address': wallet_address,
        'status': 'completed'
    }).execute()
    
    # 6. –û–±–Ω–æ–≤–∏—Ç—å –≥–ª–æ–±–∞–ª—å–Ω—É—é —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫—É (–í–ò–†–¢–£–ê–õ–¨–ù–û!)
    supabase.rpc('update_burn_stats', {
        'burned': burn_amount,
        'recycled': pool_amount
    }).execute()
    
    return {
        'success': True,
        'net_amount': net_amount,
        'signature': signature,
        'burn_amount': burn_amount,
        'pool_amount': pool_amount,
        'team_amount': team_amount
    }
